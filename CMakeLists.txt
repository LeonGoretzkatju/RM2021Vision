CMAKE_MINIMUM_REQUIRED(VERSION 3.5)
project(TJU-RM-2021)

SET(CMAKE_BUILD_TYPE RELEASE)
SET(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -O3")

FIND_PACKAGE(OpenCV REQUIRED)
